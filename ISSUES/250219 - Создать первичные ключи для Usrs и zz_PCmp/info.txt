
Создать в базе данных ES_OPI первичные ключи для Usrs и zz_PCmp.

// Установить поле NR в NOT NULL:

UPDATE RDB$RELATION_FIELDS SET RDB$NULL_FLAG = 1
WHERE RDB$FIELD_NAME = 'KDUSR' AND RDB$RELATION_NAME = 'USRS';

UPDATE RDB$RELATION_FIELDS SET RDB$NULL_FLAG = 1
WHERE RDB$FIELD_NAME = 'NR' AND RDB$RELATION_NAME = 'ZZ_PCMP';

// Создать PK:

ALTER TABLE USRS ADD PRIMARY KEY (KDUSR);

ALTER TABLE ZZ_PCMP ADD PRIMARY KEY (NR);
